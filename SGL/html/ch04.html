For many, the best way to become familiar with the capabilites of the SGL library is through examples. This chapter provides several simple examples that showcase the SGL's capabilities.

4.1. Traffic Light Simulation

Example 4.1 models a simple traffic light. The signal changes when the user depresses the mouse button over the window. The program provides a good example of drawing simple shapes and responding to user input via a pointing device.

Example 4.1. Traffic Light Simulation

#include <iostream>
#include <memory>
#include <GL/sgl.hpp>

using namespace std;
using namespace sgl;

//  Color constants used by the traffic light 
const int red    = 0,
          green  = 1,
          yellow = 2;

class TrafficLightWindow : public Window {
    int color;  //  Traffic light's current color
public:
    //  Light's initial color is red
    TrafficLightWindow() : Window("Traffic Light", 250, 600),
                color(red) {}
    void paint() override {
        //  Draw the frame
        set_color(GRAY);
        fill_rectangle(25, 25, 200, 550);

        //  Draw the red lamp
        if (color == red)
            set_color(RED);
        else
            set_color(BLACK);
        fill_circle(125, 475, 70);
        //  Draw the yellow lamp
        if (color == yellow)
            set_color(YELLOW);
        else
            set_color(BLACK);
        fill_circle(125, 300, 70);
        //  Draw the green lamp
        if (color == green)
            set_color(GREEN);
        else
            set_color(BLACK);
        fill_circle(125, 125, 70);
    }

    void mouse_pressed(double, double, MouseButton) override {
        //  Update light's current color
        color = (color + 1) % 3;
        repaint();
    }
};

int main() {
    make_shared<TrafficLightWindow>()->run();
}
