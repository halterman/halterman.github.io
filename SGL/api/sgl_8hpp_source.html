96 static void GetOGLPos(int x, int y, double *vec) {
97  GLint viewport[4];
98  GLdouble modelview[16];
99  GLdouble projection[16];
100  GLfloat winX, winY, winZ;
101  GLdouble posX, posY, posZ;
102 
103  glGetDoublev(GL_MODELVIEW_MATRIX, modelview);
104  glGetDoublev(GL_PROJECTION_MATRIX, projection);
105  glGetIntegerv(GL_VIEWPORT, viewport);
106 
107  winX = static_cast<float>(x);
108  winY = static_cast<float>(viewport[3] - y);
109  glReadPixels(x, static_cast<int>(winY), 1, 1,
110  GL_DEPTH_COMPONENT, GL_FLOAT, &winZ);
111 
112  gluUnProject(winX, winY, winZ, modelview, projection, viewport,
113  &posX, &posY, &posZ);
114 
115  vec[0] = posX; vec[1] = posY; vec[2] = posZ;
116 }

934

947  unsigned long numRows = get_long(fin); // number of rows in image
948  get_short(fin); // number of planes, always 1
949  unsigned short bitsPerPixel = get_short(fin); // 8 or 24; allow 24 here
950  get_long(fin); // compression, must be 0 for uncompressed
951  get_long(fin); // total bytes in image
952  get_long(fin); // x pels, always 0
953  get_long(fin); // y pels, always 0

955  get_long(fin); // impColors, always 0

958  std::cout << "Not a 24 bit/pixelimage, or is compressed!\n";

960  // Check for length of sides--must be a power of two
961  if (!power_of_2(numRows) || !power_of_2(numCols))
962  std::cout << "Length of sides must be powers of two\n";
963  else { // add bytes at end of each row so total # is a multiple of 4

965  int nBytesInRow = ((3 * numCols + 3)/4) * 4,
966  numPadBytes = nBytesInRow - 3 * numCols; // need this many
967

970  nCols = numCols;
971 
972  //make space for array

974 
975  if ( !pixel ) // Check for proper memory allocation
976  std::cout << "Out of memory!\n";

978  long count = 0;
979  for (int row = 0; row < nRows; row++) { // read pixel values
980  for (int col = 0; col < nCols; col++) {
981  char r, g, b;
982  // Read three bytes ...
983  fin.get(b);
984  fin.get(g);

988  pixel[count].green = g;
989  pixel[count].blue = b;
990  count++;
991  }

993  for(int k = 0 ; k < numPadBytes ; k++)
994  fin.get();
995  }

997  result = true;
998  }
999  }

1002  }
